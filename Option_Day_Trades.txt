orbAllowsPutActive   = not useORB or orbAllowsPutProfile

// â”Œâ”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”

// â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”˜

if barstate.isconfirmed or barstate.isrealtime

        if showEarlyAlerts

    if putState == EARLY and not na(putEarlyBar) and bar_index - putEarlyBar > maxEarlyBars

        if showStopLine

            // >>> LABEL: T2 <<<
            if showLabels
                label.new(bar_index, high, "T2 âœ“\n+" + str.tostring(putOptionPnl, "#.#") + "%", style=label.style_label_down, color=color.fuchsia, textcolor=color.white, size=size.tiny)
        
        // T3 Hit
        if putT2Hit and not putT3Hit and closeExec <= putEntry * (1 - t3Pct/100)
            putT3Hit := true
